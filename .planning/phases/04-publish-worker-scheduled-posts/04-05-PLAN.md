    **1. `packages/web/src/hooks/use-rate-limit.ts`** — TanStack Query hook:
    ```typescript
    export function useRateLimit(profileId: string | null) {
      return useQuery({
        queryKey: ['rate-limit', profileId],
        queryFn: () => apiClient.getRateLimit(profileId!),
        enabled: !!profileId,
        staleTime: 30_000, // 30s cache — budget changes rarely mid-session
      });
    }

    export function useUpdateRateLimit() {
      const qc = useQueryClient();
      return useMutation({
        mutationFn: ({ profileId, body }: { profileId: string; body: RateLimitUpdate }) =>
          apiClient.updateRateLimit(profileId, body),
        onSuccess: (_, vars) => {
          qc.invalidateQueries({ queryKey: ['rate-limit', vars.profileId] });
          qc.invalidateQueries({ queryKey: ['profiles'] });  // profile card re-fetches
        },
      });
    }
    ```
