    ```typescript
    import type { Job } from 'bullmq';
    import { ApiResponseError } from 'twitter-api-v2';

    const BACKOFF_SCHEDULE_MS = [30_000, 5 * 60_000, 30 * 60_000] as const;
    const MAX_BACKOFF_MS = 30 * 60_000;

    export function buildBackoffStrategy() {
      return (attemptsMade: number, _type: string, err: Error, _job: Job): number => {
        const scheduleIndex = attemptsMade - 1;

        if (err instanceof ApiResponseError && err.rateLimitError && err.rateLimit) {
          const resetMs = err.rateLimit.reset * 1000 - Date.now();
          if (resetMs > 0) {
            return Math.min(resetMs, MAX_BACKOFF_MS);
          }
        }

        return BACKOFF_SCHEDULE_MS[scheduleIndex] ?? MAX_BACKOFF_MS;
      };
    }
    ```

    **1. `packages/worker/src/publish-worker.ts`** — copy RESEARCH.md Pattern 3 with adjustments:

    ```typescript
    import { Worker, UnrecoverableError, Queue, type Job } from 'bullmq';
    import type { Redis } from 'ioredis';
    import { QUEUE_NAMES, JOB_NAMES, classifyTwitterError } from '@sms/shared';
    import { createLogger } from '@sms/shared';
    import type { WorkerDb } from './db.js';
    import { publishPost, PostLifecycleAbort } from './post-lifecycle.service.js';
    import { callTwitter } from './twitter-publish.service.js';
    import { checkTwitterBudget } from '@sms/api/services/rate-limit.service.js';
    import { buildBackoffStrategy } from './backoff.js';

    interface PublishJobPayload {
      postId: string;
      postVersion: number;
      correlationId: string;
    }

    interface PublishWorkerDeps {
      redis: Redis;
      db: WorkerDb;
      notificationQueue: Queue;
    }

    export function createPublishWorker({ redis, db, notificationQueue }: PublishWorkerDeps): Worker<PublishJobPayload> {
      const baseLogger = createLogger('publish-worker');

      const worker = new Worker<PublishJobPayload>(
        QUEUE_NAMES.publish,
        async (job) => {
          const logger = baseLogger.child({
            correlationId: job.data.correlationId,
            postId: job.data.postId,
            jobId: job.id,
            attempt: job.attemptsMade + 1,
          });

          try {
            const result = await publishPost(db, {
              postId: job.data.postId,
              expectedVersion: job.data.postVersion,
              correlationId: job.data.correlationId,
              currentAttemptNum: job.attemptsMade + 1,
              callTwitter: (profile, postText, isThread) =>
                callTwitter({ profile, postText, isThread, correlationId: job.data.correlationId }),
              checkBudget: async (profileId) => {
                const state = await checkTwitterBudget(db, { profileId, additionalPostCount: 1 });
                return { wouldExceed: state.wouldExceed };
              },
            });
            logger.info({ platformPostId: result.platformPostId }, 'Publish succeeded');
            return result;
          } catch (err) {
            if (err instanceof PostLifecycleAbort) {
              // Graceful aborts — don't retry, don't log as failure.
              if (err.reason === 'already_published') {
                logger.info('Idempotent skip — post already published');
                return { skipped: true, reason: err.reason } as unknown as { platformPostId: string };
              }
              if (err.reason === 'version_mismatch' || err.reason === 'budget_exhausted' || err.reason === 'not_scheduled') {
                logger.info({ reason: err.reason }, 'Graceful abort — will be retried by scanner');
                return { skipped: true, reason: err.reason } as unknown as { platformPostId: string };
              }
            }

            const classification = classifyTwitterError(err);
            if (classification.kind === 'permanent') {
              logger.warn({ errorCode: classification.errorCode }, 'Permanent failure — skipping retries');
              throw new UnrecoverableError(classification.message);
            }

            logger.warn({ errorCode: classification.errorCode }, 'Transient failure — will retry');
            throw err;
          }
        },
        {
          connection: redis,
          concurrency: 2,
          lockDuration: 30_000,
          stalledInterval: 30_000,
          maxStalledCount: 1,
          settings: {
            backoffStrategy: buildBackoffStrategy(),
          },
        },
      );

      // Failed listener — fires when BullMQ has given up (attemptsMade >= attempts OR UnrecoverableError).
      // Emits a notification event per WORKER-07 / D-11.
      worker.on('failed', async (job, err) => {
        if (!job) return;
        const isFinalFailure =
          err.name === 'UnrecoverableError' || (job.attemptsMade >= (job.opts.attempts ?? 4));
        if (!isFinalFailure) return;

        try {
          await notificationQueue.add(JOB_NAMES.publishFailedNotification, {
            kind: 'publish_failed',
            postId: job.data.postId,
            correlationId: job.data.correlationId,
            reason: err.message,
            at: new Date().toISOString(),
          });
        } catch (enqueueErr) {
          baseLogger.error({ err: enqueueErr, postId: job.data.postId }, 'Failed to enqueue notification event');
        }
      });

      return worker;
    }
    ```
