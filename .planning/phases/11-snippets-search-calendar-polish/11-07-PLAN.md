       Factory:
           interface CalendarDependencies { db: Db; }
           export function createCalendarRouter({ db }: CalendarDependencies): Router {
             const router = Router();
             const logger = createLogger('calendar-route');
             router.get('/api/calendar', requireAuth, async (req, res) => {
               const parsed = calendarQuerySchema.safeParse(req.query);
               if (!parsed.success) {
                 res.status(400).json({ error: 'Validation failed', details: parsed.error.issues });
                 return;
               }
               const { from, to, scope, platforms, profileIds, tagIds, search } = parsed.data;
               const userId = req.session.userId!;
               const conditions = [
                 eq(posts.userId, userId),
                 isNotNull(posts.scheduledAt),
                 gte(posts.scheduledAt, new Date(from)),
                 lte(posts.scheduledAt, new Date(to)),
               ];
               // Scope filter
               if (scope === 'scheduled') {
                 conditions.push(eq(posts.status, 'scheduled'));
               } else if (scope === 'queued') {
                 conditions.push(eq(posts.status, 'queued'));
               } else {
                 // 'both'
                 conditions.push(inArray(posts.status, ['scheduled', 'queued', 'publishing']));
               }
               // Platform / profile / tag filters
               if (platforms?.length) conditions.push(inArray(posts.platform, platforms));
               if (profileIds?.length) conditions.push(inArray(posts.profileId, profileIds));
               // Tag filter requires EXISTS subquery against post_tags+tags
               if (tagIds?.length) {
                 conditions.push(sql`EXISTS (SELECT 1 FROM post_tags pt WHERE pt.post_id = ${posts.id} AND pt.tag_id = ANY(${tagIds}))`);
               }
               // Optional search reuses the FTS pattern from plan 06
               if (search) {
                 const tsq = sql`plainto_tsquery('english', ${search})`;
                 conditions.push(sql`(${posts.searchVector} || ${posts.tagSearchVector}) @@ ${tsq}`);
               }
               const rows = await db
                 .select({
                   id: posts.id,
                   platform: posts.platform,
                   profileId: posts.profileId,
                   profileDisplayName: socialProfiles.displayName,
                   status: posts.status,
                   scheduledAt: posts.scheduledAt,
                   text: posts.text,
                 })
                 .from(posts)
                 .leftJoin(socialProfiles, eq(posts.profileId, socialProfiles.id))
                 .where(and(...conditions));
               // Annotate hasConflict per row using existing checkConflicts (D-13).
               const events = await Promise.all(rows.map(async (row) => {
                 const conflicts = await checkConflicts(db, userId, row.profileId, row.scheduledAt!.toISOString(), row.id);
                 return {
                   id: row.id,
                   platform: row.platform,
                   profileId: row.profileId,
                   profileDisplayName: row.profileDisplayName ?? '',
                   status: row.status,
                   scheduledAt: row.scheduledAt!.toISOString(),
                   textPreview: (row.text ?? '').slice(0, 60),
                   hasConflict: conflicts.length > 0,
                 };
               }));
               res.json({ events });
             });
             return router;
           }

       Naming per CLAUDE.md §Naming: `hasConflict` (boolean prefix); no generic `data`/`result` names. Per CLAUDE.md §Type Safety: explicit return type on `createCalendarRouter`; no `any` for the rows.
