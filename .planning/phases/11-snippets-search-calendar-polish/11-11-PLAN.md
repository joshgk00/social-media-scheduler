           import { useQuery } from '@tanstack/react-query';
           import { apiClient } from '../lib/api-client.js';
           import type { CalendarResponse, CalendarQuery } from '@sms/shared';

           export function useCalendarPosts(query: CalendarQuery | undefined) {
             const params = new URLSearchParams();
             if (query) {
               params.set('from', query.from);
               params.set('to', query.to);
               params.set('scope', query.scope);
               query.platforms?.forEach((p) => params.append('platforms', p));
               query.profileIds?.forEach((id) => params.append('profileIds', id));
               query.tagIds?.forEach((id) => params.append('tagIds', id));
               if (query.search) params.set('search', query.search);
             }
             const queryString = params.toString();
             return useQuery({
               queryKey: ['calendar', query],
               queryFn: () => apiClient.get<CalendarResponse>(`/api/calendar?${queryString}`),
               enabled: !!query,
               staleTime: 30_000,
             });
           }

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Server `/api/calendar` events → client | Event payload includes `textPreview` and `profileDisplayName`; rendered as text (React-escaped) |
| User filter inputs → URL → server | Filter values bounded by Zod schema (plan 01) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-11-11-01 | Tampering (XSS via textPreview in calendar entry) | Calendar event render | mitigate | Event title is a plain string interpolation `${platformAbbrev(e.platform)} · ${e.textPreview}`; rendered by rbc as React text — escaped by default. No `dangerouslySetInnerHTML` |
| T-11-11-02 | DoS (huge date range from manipulated URL) | useCalendarPosts | mitigate | Server-side window cap of 100 days (plan 01 Zod refine) returns 400; client also clamps via `normalizeRange` outputs which always come from the rbc current view (max ~35-day month) |
| T-11-11-03 | UX (calendar appears empty on Week/Day views) | onRangeChange | mitigate | Pitfall 5: `normalizeRange` handles all three shapes (object, week array, day array); Test F asserts each |
| T-11-11-04 | Information Disclosure (rbc CSS leaks across app) | global stylesheet | mitigate | Pitfall 3: import rbc CSS only inside CalendarPage module; `rg` check in <done> verifies single import location |
| T-11-11-05 | Tampering (CSRF — no risk; route is GET-only) | /api/calendar | accept | Read-only endpoint; CSRF only protects state-changing requests |
| T-11-11-06 | Logic flaw (conflict logic diverges from creation form) | hasConflict | mitigate | Server-computed by reusing `checkConflicts` (plan 07 D-13); client only renders the boolean — divergence structurally impossible |
</threat_model>
