       import { describe, it, expect } from 'vitest';
       import { z } from 'zod';
       import * as bulkJobs from '@sms/shared/schemas/bulk-jobs';
       import * as bulkImport from '@sms/shared/schemas/bulk-import';
       import * as bulkNotifs from '@sms/shared/schemas/bulk-notifications';
       import * as notifs from '@sms/shared/schemas/notifications';

       const FORBIDDEN_KEY_RE = /openai|api[_-]?key/i;

       /**
        * Unwrap ZodEffects / ZodOptional / ZodDefault wrappings to reach the
        * underlying ZodObject so `_def.shape()` works (RESEARCH Assumption A4).
        */
       function unwrapToObject(schema: z.ZodTypeAny): z.ZodObject<any> | null {
         let current: any = schema;
         for (let depth = 0; depth < 10; depth++) {
           if (current?._def?.typeName === 'ZodObject') return current;
           if (current?._def?.schema) { current = current._def.schema; continue; }
           if (current?._def?.innerType) { current = current._def.innerType; continue; }
           return null;
         }
         return null;
       }

       function getFieldNames(schema: z.ZodTypeAny): string[] {
         const obj = unwrapToObject(schema);
         if (!obj) return [];
         const shape = typeof obj._def.shape === 'function' ? obj._def.shape() : obj._def.shape;
         return Object.keys(shape ?? {});
       }

       const allModules = { bulkJobs, bulkImport, bulkNotifs, notifs };
       const allSchemaEntries: Array<[string, z.ZodTypeAny]> = [];
       for (const [moduleName, mod] of Object.entries(allModules)) {
         for (const [exportName, value] of Object.entries(mod as Record<string, unknown>)) {
           if (value && typeof value === 'object' && '_def' in (value as object)) {
             allSchemaEntries.push([`${moduleName}.${exportName}`, value as z.ZodTypeAny]);
           }
         }
       }

       describe('SEC-07: BullMQ job-data schemas reject OpenAI/api-key fields', () => {
         it('discovers at least one job-data schema (sanity check)', () => {
           expect(allSchemaEntries.length).toBeGreaterThan(0);
         });

         it.each(allSchemaEntries)('%s declares no OpenAI/api-key field', (_label, schema) => {
           for (const fieldName of getFieldNames(schema)) {
             expect(fieldName, `field "${fieldName}" matches forbidden pattern`).not.toMatch(FORBIDDEN_KEY_RE);
           }
         });
       });
       ```

Per CLAUDE.md §Testing: assert preconditions first (the sanity-check `it` ensures the test would fail loudly if imports break and silently zero-out the assertion set).

       # Security Policy

       ## Reporting a Vulnerability

       This is a self-hosted single-user application. If you find a security issue, open a GitHub issue with the label `security` or contact the repository owner directly.

       ## Sensitive Credentials

       ### OAuth Tokens
       All Twitter/LinkedIn/Facebook OAuth tokens are encrypted at rest using AES-256-GCM with a key supplied via the `ENCRYPTION_KEY` environment variable. The key is never persisted to the database, source control, Redis, or logs. See `packages/shared/src/encryption/` for the implementation.

       ### Session Cookies
       Sessions use HTTP-only Secure cookies with `SameSite=Strict`. Session storage is Redis-backed via `connect-redis`. CSRF protection is enforced on all state-changing requests via the `csrf-csrf` Double-Submit Cookie pattern.

       ## OpenAI API Key Handling (SEC-07)

       The OpenAI API key (used by future AI generation features) follows a strict per-request handling policy:

       1. **Per-request only.** The key is supplied in the request body of a single AI endpoint call. It is never read from environment variables on the server, and there is no persistent key store.
       2. **Never persisted.** The key is never written to the database, Redis, BullMQ job payloads, files on disk, or any audit trail.
       3. **Never logged.** The Pino logger config in `packages/shared/src/logger.ts` redacts the keys `openai_api_key`, `openaiApiKey`, and `OPENAI_API_KEY` at any nesting depth before serialization. Wildcards cover nested object positions; the request body and request headers are explicitly redacted.
       4. **Never queued.** A static Vitest contract test (`packages/api/src/__tests__/sec-07-job-schema.test.ts`) enumerates every BullMQ job-data Zod schema and fails CI if any schema field name matches the regex `/openai|api[_-]?key/i`. This blocks future code from accidentally embedding the key in a queued job.
       5. **Per-call lifetime.** The key lives only on the call stack of the AI generation handler — read from `req.body`, passed to the OpenAI client, and discarded when the function returns. No caching, no closures retaining the value.

       This policy is enforced by:

       - The redaction config at `packages/shared/src/logger.ts` (`DEFAULT_REDACT.paths`).
       - The contract test at `packages/api/src/__tests__/sec-07-job-schema.test.ts`.
       - The unit tests at `packages/api/src/__tests__/logger.test.ts` covering the three key-name variants and the wildcard nested-object path.

       Phase 11 ships these guardrails ahead of any AI feature so Phase 12 (AI Post Generation) inherits the policy by default.

       ## Encryption Key Rotation

       The `ENCRYPTION_KEY` rotates without user re-authentication via the `token_encryption_version` column on the `social_profiles` table. The migration runs decrypt-with-old, re-encrypt-with-new in batches; see `packages/api/src/services/oauth.service.ts`.
       ```
