```typescript
export async function runMigrations(databaseUrl: string, logger: MigrationLogger = defaultLogger) {
  const client = postgres(databaseUrl, { max: 1 });
  try {
    const reserved = await client.reserve();         // Pin one backend PID
    try {
      await ensureMigrationsTable(reserved);
      await reserved.unsafe(`SELECT pg_advisory_lock(${MIGRATION_LOCK_KEY}::bigint)`);  // H-01 fix
      try {
        // ... journal read + migration loop with applyMigration({ client: reserved, ... }) ...
      } finally {
        // unlock on same session that holds the lock
        await reserved.unsafe(`SELECT pg_advisory_unlock(${MIGRATION_LOCK_KEY}::bigint)`);
      }
    } finally {
      await reserved.release();
    }
  } finally {
    await client.end();
  }
}
```

```typescript
async function applyMigration({ client, entry, sql, hash, logger }) {
  await client.begin(async (tx) => {          // H-02 fix: atomic per-migration transaction
    for (const statement of statements) {
      try { await tx.unsafe(statement); }
      catch (err) {
        // D-05: duplicate codes swallowed inside tx (tx still commits)
        // All other codes rethrow → tx aborts → partial DDL rolled back
      }
    }
    await tx`INSERT INTO "drizzle"."__drizzle_migrations" ...`;    // Inside transaction
  });
  logger.info('Migration applied', { ... });   // Fires only after commit
}
```

## Advisory Lock Resource Cleanup Order

Outside-in nesting:
1. `pg_advisory_unlock` fires first (innermost finally) — on the same reserved session that acquired it
2. `reserved.release()` fires second (middle finally) — connection returned to pool
3. `client.end()` fires last (outer finally) — pool torn down

This order is mandatory. If `reserved.release()` ran before `pg_advisory_unlock`, the lock would still release (session close auto-releases) but with a noisy connection-closed error — Pitfall 2 from RESEARCH.md.
